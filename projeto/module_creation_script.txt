#integração fontawesome
    app npm i @fortawesome/angular-fontawesome
    app npm i @fortawesome/fontawesome-svg-core
    app npm i @fortawesome/free-solid-svg-icons
    app npm i @fortawesome/free-regular-svg-icons
    app npm i @fortawesome/free-brands-svg-icons

    #em src/app/app.module.ts
    import { FontAwesomeModule, FaIconLibrary } from '@fortawesome/angular-fontawesome';
    import { fas } from '@fortawesome/free-solid-svg-icons'
    import { far } from '@fortawesome/free-regular-svg-icons'
    import { fab } from '@fortawesome/free-brands-svg-icons'

    # adicionar o construtor na classe AppModule
    constructor(library: FaIconLibrary) { 
		library.addIconPacks(fas, fab, far);
	}

    #adicionar FontAwesomeModule ao imports
